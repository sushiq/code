
def greet():
    print("Hello! My name is Chatty Bot.")
    print("I was created to assist you.")


def get_name():
    print("Please, tell me your name.")
    name = input()
    print("Nice to meet you,", name)


def guess_age():
    print("Let me guess your age.")
    print("Enter remainders of dividing your age by 3, 5 and 7.")

    r3 = int(input())
    r5 = int(input())
    r7 = int(input())

    age = (r3 * 70 + r5 * 21 + r7 * 15) % 105
    print("Your age is", age, "; that's a great age!")


def count_numbers():
    print("Now I will count to any number you want.")
    num = int(input())

    for i in range(num + 1):
        print(i)


def quiz():
    print("Let's test your programming knowledge.")
    print("Why do we use functions?")
    print("1. To repeat statements")
    print("2. To divide program into smaller parts")
    print("3. To stop program execution")
    print("4. To display output")

    answer = 2
    while True:
        user = int(input())
        if user == answer:
            print("Correct!")
            break
        else:
            print("Try again.")


def end():
    print("Congratulations! Have a nice day!")


# Main execution
greet()
get_name()
guess_age()
count_numbers()
quiz()
end()
